Step 1: Basic Tkinter Window

import tkinter as tk
from tkinter import ttk

class FaceRecognitionApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Face Recognition System")
        self.root.geometry("800x600")
        
        # Create widgets
        self.title_label = ttk.Label(root, text="Face Recognition Attendance",
                                     font=('Arial', 24, 'bold'))
        self.title_label.pack(pady=20)
        
        # Start button
        self.start_btn = ttk.Button(root, text="Start Recognition",
                                    command=self.start_recognition)
        self.start_btn.pack(pady=10)
    
    def start_recognition(self):
        print("Recognition started!")

if __name__ == "__main__":
    root = tk.Tk()
    app = FaceRecognitionApp(root)
    root.mainloop()

Step 2: Embed OpenCV Video

import cv2
from PIL import Image, ImageTk
import threading

class VideoFrame(ttk.Frame):
    def __init__(self, parent):
        super().__init__(parent)
        
        # Canvas untuk video
        self.canvas = tk.Canvas(self, width=640, height=480)
        self.canvas.pack()
        
        self.cap = None
        self.running = False
    
    def start_video(self):
        self.cap = cv2.VideoCapture(0)
        self.running = True
        
        # Run in separate thread
        thread = threading.Thread(target=self.update_frame)
        thread.daemon = True
        thread.start()
    
    def update_frame(self):
        while self.running:
            ret, frame = self.cap.read()
            if ret:
                # Convert BGR to RGB
                frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                
                # Convert to PIL Image
                img = Image.fromarray(frame_rgb)
                imgtk = ImageTk.PhotoImage(image=img)
                
                # Update canvas
                self.canvas.create_image(0, 0, anchor='nw', image=imgtk)
                self.canvas.imgtk = imgtk  # Keep reference
    
    def stop_video(self):
        self.running = False
        if self.cap:
            self.cap.release()
