Step 1: Design Capture System

import cv2
import os
from datetime import datetime

def capture_dataset(person_name, num_images=50):
    # Create folder untuk person
    folder = f"dataset/{person_name}"
    os.makedirs(folder, exist_ok=True)
    
    cap = cv2.VideoCapture(0)
    count = 0
    
    print(f"Capturing {num_images} images untuk {person_name}")
    print("Press SPACE untuk capture, ESC untuk quit")
    
    while count < num_images:
        ret, frame = cap.read()
        if not ret:
            break
            
        # Display frame dengan counter
        cv2.putText(frame, f"Captured: {count}/{num_images}", 
                   (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 
                   1, (0, 255, 0), 2)
        cv2.imshow('Capture Dataset', frame)
        
        key = cv2.waitKey(1)
        if key == 32:  # SPACE
            filename = f"{folder}/{person_name}_{count+1}.jpg"
            cv2.imwrite(filename, frame)
            count += 1
        elif key == 27:  # ESC
            break
    
    cap.release()
    cv2.destroyAllWindows()
    print(f"Captured {count} images!")
