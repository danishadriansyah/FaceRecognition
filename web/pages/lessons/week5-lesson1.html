RecognitionService Architecture

from face_detector import FaceDetector
from face_recognizer import FaceRecognizer
from dataset_manager import DatasetManager
import logging

class RecognitionService:
    def __init__(self, config):
        self.detector = FaceDetector()
        self.recognizer = FaceRecognizer(tolerance=0.6)
        self.dataset_manager = DatasetManager()
        
        # Setup logging
        logging.basicConfig(level=logging.INFO)
        self.logger = logging.getLogger(__name__)
    
    def load_dataset(self, dataset_path):
        """Load all known faces from dataset"""
        self.logger.info(f"Loading dataset from {dataset_path}")
        persons = self.dataset_manager.load_all_persons(dataset_path)
        
        for person_name, images in persons.items():
            for img in images:
                encoding = self.recognizer.encode_face(img)
                if encoding is not None:
                    self.recognizer.add_known_face(person_name, encoding)
        
        self.logger.info(f"Loaded {len(persons)} persons")
    
    def recognize_from_frame(self, frame):
        """Detect and recognize faces in frame"""
        # Step 1: Detect faces
        faces = self.detector.detect_faces(frame)
        
        results = []
        for (x, y, w, h) in faces:
            # Step 2: Extract face region
            face_img = frame[y:y+h, x:x+w]
            
            # Step 3: Generate encoding
            encoding = self.recognizer.encode_face(face_img)
            
            # Step 4: Recognize face
            if encoding is not None:
                name, confidence = self.recognizer.recognize_face(encoding)
                results.append({
                    'bbox': (x, y, w, h),
                    'name': name,
                    'confidence': confidence
                })
        
        return results
