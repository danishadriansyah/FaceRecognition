üìù Tutorial

Create image_utils.py Module

import cv2
import numpy as np

def load_image(path, grayscale=False):
    """Load image from file path"""
    img = cv2.imread(path)
    if img is None:
        raise FileNotFoundError(f"Image not found: {path}")
    if grayscale:
        img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    return img

def save_image(path, image):
    """Save image to file"""
    success = cv2.imwrite(path, image)
    if not success:
        raise IOError(f"Failed to save: {path}")
    return success

def resize_image(image, width=None, height=None):
    """Resize with aspect ratio preservation"""
    h, w = image.shape[:2]
    if width is None and height is None:
        return image
    if width is None:
        ratio = height / h
        width = int(w * ratio)
    else:
        ratio = width / w
        height = int(h * ratio)
    return cv2.resize(image, (width, height))

def display_image(title, image, wait=True):
    """Display image with window title"""
    cv2.imshow(title, image)
    if wait:
        cv2.waitKey(0)
        cv2.destroyAllWindows()

def validate_image(image):
    """Check if valid numpy array"""
    if not isinstance(image, np.ndarray):
        return False
    if image.size == 0:
        return False
    return True

Usage Example

from image_utils import *

# Load & resize
img = load_image('photo.jpg')
resized = resize_image(img, width=800)

# Validate & save
if validate_image(resized):
    save_image('photo_resized.jpg', resized)
